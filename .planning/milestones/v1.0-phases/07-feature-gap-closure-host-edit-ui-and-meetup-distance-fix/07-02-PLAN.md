    mutationFn: ({ id, data }: { id: number; data: { name?: string; email?: string | null } }) =>
      api.put<{ host: Host }>(`/api/hosts/${id}`, data),

      queryClient.invalidateQueries({ queryKey: ['hosts'] });
    },

}
