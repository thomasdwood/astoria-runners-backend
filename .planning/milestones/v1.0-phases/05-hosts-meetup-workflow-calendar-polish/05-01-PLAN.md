```typescript
import { pgTable, serial, varchar, integer, timestamp } from 'drizzle-orm/pg-core';
import { InferSelectModel, InferInsertModel, relations } from 'drizzle-orm';
import { users } from './users.js';
import { events } from './events.js';
import { recurringTemplates } from './recurringTemplates.js';

export const hosts = pgTable('hosts', {
  id: serial('id').primaryKey(),
  name: varchar('name', { length: 100 }).notNull(),
  email: varchar('email', { length: 255 }),
  userId: integer('user_id').references(() => users.id, { onDelete: 'set null' }),
  version: integer('version').notNull().default(0),
  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp('updated_at', { withTimezone: true }).defaultNow().notNull(),
});

export const hostsRelations = relations(hosts, ({ many }) => ({
  events: many(events),
  recurringTemplates: many(recurringTemplates),
}));

export type Host = InferSelectModel<typeof hosts>;
export type NewHost = InferInsertModel<typeof hosts>;
```

**NOTE on circular imports:** hosts.ts imports from events.ts, and events.ts will import from hosts.ts. To avoid circular imports, do NOT import events/recurringTemplates in hosts.ts for the relation definition. Instead, define hostsRelations as a simple object without importing the related tables, or use a forward reference pattern. The safest approach: define `hostsRelations` in hosts.ts only referencing `events` and `recurringTemplates` by importing them — but since they'll also import hosts, this will be circular. Resolution: define `hostsRelations` using lazy relation callbacks if needed, OR simply omit the reverse relations from hosts.ts (the many-side) and only define the one-side in events.ts and recurringTemplates.ts. Keep hosts.ts clean with no relational imports.
