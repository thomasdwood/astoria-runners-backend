    **Changes to meetup-export-popover.tsx:**

    1. Import `useCreateEvent` hook at the top alongside the existing `useUpdateMeetupUrl` import:
       ```
       import { useCreateEvent, useUpdateMeetupUrl } from '@/hooks/use-events';
       ```

    2. Inside the component, add the mutation:
       ```
       const createEvent = useCreateEvent();
       ```

    3. Replace the `isDbEvent` gate on the URL section with a condition that includes virtual instances:
       Change: `{isDbEvent && (`
       To: `{(isDbEvent || !!calendarEvent) && (`

    4. Update `handleSaveUrl` to handle virtual instances:
       Replace the current function body:
       ```typescript
       async function handleSaveUrl() {
         try {
           let targetId = eventId;

           if (!targetId && calendarEvent) {
             // Materialize the virtual recurring instance on-demand before saving the URL
             const result = await createEvent.mutateAsync({
               routeId: calendarEvent.routeId!,
               startDateTime: calendarEvent.startDateTime,
               startLocation: calendarEvent.startLocation ?? undefined,
               endLocation: calendarEvent.endLocation ?? undefined,
               notes: calendarEvent.notes ?? undefined,
               recurringTemplateId: calendarEvent.recurringTemplateId !== null ? calendarEvent.recurringTemplateId : undefined,
               hostId: calendarEvent.hostId ?? null,
             });
             targetId = result.event.id;
           }

           if (!targetId) return;

           await updateMeetupUrl.mutateAsync({ id: targetId, meetupUrl: urlInput || null });
           setUrlSaved(true);
           toast.success(urlInput ? 'Meetup URL saved' : 'Meetup URL cleared');
           setTimeout(() => setUrlSaved(false), 2000);
         } catch {
           toast.error('Failed to save Meetup URL');
         }
       }
       ```

       Change: `disabled={updateMeetupUrl.isPending}`
       To: `disabled={updateMeetupUrl.isPending || createEvent.isPending}`

    Check the CalendarEvent type to confirm fields like `routeId`, `startLocation`, `endLocation`, `notes`, `hostId` are available. If `hostId` is not on CalendarEvent, omit it (don't add a field that doesn't exist on the type).
